-- Script para duplicar brainrots en "Steal a Brainrot"

local function duplicateBrainrots()
    local player = game.Players.LocalPlayer
    print("Jugador encontrado: " .. tostring(player))

    local character = player.Character or player.CharacterAdded:Wait()
    print("Personaje encontrado: " .. tostring(character))

    local backpack = player:FindFirstChildOfClass("Backpack")
    print("Backpack encontrado: " .. tostring(backpack))

    if backpack then
        local brainrotFolder = backpack:FindFirstChild("BrainrotFolder")
        print("BrainrotFolder encontrado: " .. tostring(brainrotFolder))

        if brainrotFolder then
            for _, brainrot in ipairs(brainrotFolder:GetChildren()) do
                if brainrot:IsA("Tool") then
                    local clone = brainrot:Clone()
                    clone.Parent = brainrotFolder
                    print("Brainrot clonado: " .. brainrot.Name)
                end
            end
        else
            print("BrainrotFolder no encontrado en el backpack.")
        end
    else
        print("Backpack no encontrado.")
    end
end

duplicateBrainrots()
